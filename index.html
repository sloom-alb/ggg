function doGet(e) { // احصل على الورقة المطلوبة من جدول البيانات var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('درجات الأسبوع (حفظ)'); // استبدل "Sheet1" باسم ورقتك // احصل على جميع البيانات الموجودة في الورقة var data = sheet.getDataRange().getValues(); // تهيئة مصفوفة لتخزين البيانات var jsonData = []; // معالجة كل صف في الورقة وتحويله إلى JSON for (var i = 0; i < data.length; i++) { jsonData.push({ column1: data[i][0], // العمود الأول column2: data[i][0], // العمود الثاني column3: data[i][0], // العمود الثالث column4: data[i][0], // العمود الربع column5: data[i][0], // العمود الخامس column6: data[i][0], // العمود السادس column7: data[i][1], //العمود السابع // يمكنك إضافة المزيد من الأعمدة حسب الحاجة }); } // إرجاع البيانات على هيئة JSON return ContentService.createTextOutput(JSON.stringify(jsonData)).setMimeType(ContentService.MimeType.JSON); }
